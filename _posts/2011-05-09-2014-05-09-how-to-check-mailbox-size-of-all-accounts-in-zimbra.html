Open you VIM to write down below script and execute it.Â 

#!/bin/bash 
all_account=`zmprov gaa`
for account in ${all_account} do mb_size=`zmmailbox -z -m ${account} gms`
echo "Mailbox size of ${account} = ${mb_size}"
done
